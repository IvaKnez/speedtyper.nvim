INSTALLATION                    *speedtyper.nvim-speedtyper.nvim-installation*

packer <https://github.com/wbthomason/packer.nvim>

>lua
    use {
        "speedtyper.nvim",
        branch = "main",
        config = function()
            require('speedtyper').setup({
                -- your config
            })
        end
    }
<

lazy <https://github.com/folke/lazy.nvim>

>lua
    {
        "speedtyper.nvim",
        branch = "main",
        opts = {
            -- your config
        }
    }
<

  |speedtyper.default_opts|
>lua
    {
        time = 30,
        window = {
            height = 0.15, -- integer grater than 0 or float in range (0, 1)
            width = 0.55, -- integer grater than 0 or float in range (0, 1)
            border = "rounded", -- "none" | "single" | "double" | "rounded" | "shadow" | "solid"
        }
    }

COMMANDS                            *speedtyper.nvim-speedtyper.nvim-commands*

  -----------------------------------------------------------------------
  Command                Description
  ---------------------- ------------------------------------------------
  :Speedtyper <time>     Runs typing test with <time> seconds on the
                         clock.

  :Speedtyper            Runs typing test with default time setting on
                         the clock.
  -----------------------------------------------------------------------
